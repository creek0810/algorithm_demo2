Name: Bubble sort

Time complexity: O(x2)

Space complexity: Θ(1)

Data structure: Array

Detail: Stable sort

Code


void bubble_sort(vector&array){
    int size =array.size();
    for(int i = 1; i < size; i++){
        for(int j = 0; j < size - i - 1; j++){
            if(array[j] > array[j+1]){
                swap(array[j], array[j+1]);
            }
        }
    }
}
