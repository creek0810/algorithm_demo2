function bubble_sort (array, length) {
　　var i, j;
　　for(i from 0 to length-1){
　　　　for(j from 0 to length-1-i){
　　　　 　　if (array[j] > array[j+1])
 　　　　　　　　swap(array[j], array[j+1])
 　　　　}
　　}
}
