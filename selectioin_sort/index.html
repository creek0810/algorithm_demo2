function selection_sort (array, length) {
　　var i, j, min;
　　for(i from 0 to length-2){
　　　　min = i;
　　　　for(j from i+1 to length-1){
　　　　 　　if (array[min] > array[j])
 　　　　　　　　min = j
　　　　}
　　　　 swap(array[min],array[j]); 　　　　
　　}
}
